#### 1. Initialize the Passkey Client

<CodeBlock language="typescript">
  {`import { Turnkey } from "@turnkey/sdk-browser";`}
  {`\n\nconst turnkey = new Turnkey({
  apiBaseUrl: "https://api.turnkey.com",
  defaultOrganizationId: process.env.TURNKEY_ORGANIZATION_ID
})`}
  {`\nconst passkeyClient = turnkey.passkeyClient();`}
</CodeBlock>

#### 2. Initialize an Ethers Provider and Turnkey Signer

<CodeBlock language="typescript">
  {`import { ethers } from "ethers";`}
  {`\nimport { TurnkeySigner } from "@turnkey/ethers";`}
  {`\n\nconst provider = new ethers.JsonRpcProvider(<provider api url>);`}
  {`\nconst currentUser = await turnkey.getCurrentUser();`}
  {`\nconst turnkeySigner = new TurnkeySigner({
  client: passkeyClient,
  organizationId: currentUser.organization.organizationId,
  signWith: "<wallet address to sign with>"
})`}
  {`\nconst connectedSigner = turnkeySigner.connect(provider);`}
</CodeBlock>

#### 3. Call `sendTransaction` with the Turnkey Signer

<CodeBlock language="typescript">
  {`const transactionRequest = {
  to: "<destination address>",
  value: ethers.parseEther("<amount to send>"),
  type: 2
}`}
  {`\nconst sendTransaction = await connectedSigner.sendTransaction(transactionRequest);`}
</CodeBlock>
